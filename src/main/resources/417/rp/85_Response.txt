function resetUserDiv()

    var prodDivH = document.getElementById("prodDiv").scrollHeight;
    var checkH = document.getElementById("check").scrollHeight;
    var newH = bodyH - prodDivH - checkH - 32;
    
    var userDiv = document.getElementById("userProductListDiv");

    {
        userDiv.style.height = newH;
    }
    else

	function appendTokenToUrl(url){
		try{
			if(undefined == document.all["hiddenTokenName"]){
				return url;
			}
			var hiddenTokenNameSave = document.all["hiddenTokenName"].value;
			if(null ==  hiddenTokenNameSave || ""== hiddenTokenNameSave){
				return url;
			}
			var hiddenTokenNameValue= document.all[""+hiddenTokenNameSave].value;
			if(null == hiddenTokenNameValue || ""== hiddenTokenNameValue){
				return url;
			}
                  var tmpUrl = url;
			if(url.indexOf("?") == -1){
				tmpUrl+="?";
			}else{
				tmpUrl+="&";
			}
			
			tmpUrl+="hiddenTokenName="+hiddenTokenNameSave;
			tmpUrl+="&"+hiddenTokenNameSave+"="+hiddenTokenNameValue;
                  return tmpUrl;
			}catch(e){
				return url;
			}
	}
